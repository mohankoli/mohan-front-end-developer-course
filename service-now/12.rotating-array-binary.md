## JavaScript Implementation

```javascript
function search(nums, target, k){

    let sorted = nums.slice(k).concat(nums.slice(0,k))

    let left = 0
    let right = sorted.length - 1

    while(left <= right){

        let mid = Math.floor((left + right)/2)

        if(sorted[mid] === target){
            return mid
        }

        if(sorted[mid] < target){
            left = mid + 1
        }else{
            right = mid - 1
        }
    }

    return -1
}
